🔧 Handler Implementations

1. Pacman Handler

install_pacman() {
    local file="$1"
    log "Installing pacman package: $file"
    sudo pacman -U "$file"
    # Creates package entry in /var/lib/pacman/local/
}

2. tar.xz Portable Handler

install_tarportable() {
    local file="$1"
    local name="${file%.*}"
    name="${name##*/}"
    
    mkdir -p "$HOME/.local/opt/$name"
    tar -xf "$file" -C "$HOME/.local/opt/$name"
    
    # Find executable and symlink
    find "$HOME/.local/opt/$name" -type f -executable | head -1 | \
        xargs -I{} ln -sf {} "$HOME/.local/bin/$name"
}

3. .deb Handler

install_debtap() {
    local file="$1"
    log "Installing .deb via debtap: $file"
    debtap "$file" || echo "[WARN] Ensure debtap is installed: yay -S debtap"
}

4. AppImage Handler

install_appimage() {
    local file="$1"
    local name="${file##*/}"
    name="${name%.*}"
    
    cp "$file" "$HOME/.local/bin/$name"
    chmod +x "$HOME/.local/bin/$name"
}

🚀 Deployment

# Local install
make install

# AUR package
updpkgopts PKGBUILD && makepkg --printsrcinfo > .SRCINFO
git add PKGBUILD .SRCINFO && git commit -m "Release v1.0.0"
makepkg --sync --install

# GitHub release
git tag v1.0.0 && git push origin v1.0.0
# GitHub Actions auto-builds and creates release
